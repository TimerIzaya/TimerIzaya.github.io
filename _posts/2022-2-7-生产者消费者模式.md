## 前言

第一次派上实际用处的设计模式，在多线程IO读写的时候，近乎标准答案一样的存在。  



## 设计思想

以IO读写为例，**很多时候多线程读来的数据并不一定要具体某个线程去完成写操作**（这就是我第一次多线程IO犯下的错），**任何一个读线程读完了都可以放进中间队列，任何写线程都可以从中拿**。类似于进程之间的管道通信，使用第三方的共享数据结构来保证读写线程之间的均衡。

另外提一嘴，如果某个线程读来的数据需要另一个指定线程去完成写，那就是**保护性暂停模式**，也就是join方法的设计模式。  



## 代码实现

```java
package com.timer;

import java.util.LinkedList;
import java.util.concurrent.TimeUnit;

class Message {

    private int id;

    private String value;

    public Message(int id, String value) {
        this.id = id;
        this.value = value;
    }

    public int getId() {
        return id;
    }

    public String getValue() {
        return value;
    }

    @Override
    public String toString() {
        return "Message{" +
                "id=" + id +
                ", value='" + value + '\'' +
                '}';
    }
}

//中间消息队列
class MessageQueue {

    //双向链表来表示生产者和消费者之间的联系通道
    private LinkedList<Message> list = new LinkedList<>();

    //队列长度为5
    private int cap = 5;

    //消费者从队列中获得消息
    public Message get() {
        //保证消费者是线程安全的
        synchronized (list) {
            //保证此时队列是不为空的
            while (list.isEmpty()) {
                try {
                    System.out.println("线程: " + Thread.currentThread().getName() + " 由于队列为空，正在等待获取...");
                    list.wait();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            //如果队列不为空，返回并删除最前端的值。
            Message message = list.removeFirst();
            System.out.println("线程: " + Thread.currentThread().getName() + " 获得 " + message.toString());
            //唤醒另一边因为队列满了而阻塞的生产者
            list.notifyAll();
            return message;
        }
    }

    //生产者往队列中放消息
    public void put(Message message) {
        //保证生产者的线程安全
        synchronized (list) {
            //如果队列满了，则进入阻塞
            while (list.size() == cap) {
                try {
                    System.out.println("线程: " + Thread.currentThread().getName() + " 由于队列已满，正在等待写入...");
                    list.wait();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            System.out.println("线程: " + Thread.currentThread().getName() + " 写入 " + message.toString());
            //如果队列不满，则把值加入队列末端。
            list.addLast(message);
            //同时唤醒因为队列为空而阻塞的生产者
            list.notifyAll();
        }
    }
}

public class ConsumerAndWorker {
    private int workThreadNum;

    public static void main(String[] args) {
        MessageQueue messageQueue = new MessageQueue();
        for (int i = 0; i < 10; i++) {
            int id = i;
            new Thread(() -> {
                messageQueue.put(new Message(id, "message" + id));
            }, Integer.toString(i)).start();
            sleep(1);
        }

        for (int i = 0; i < 10; i++) {
            new Thread(() -> {
                messageQueue.get();
            }, Integer.toString(i)).start();
            sleep(1);
        }
    }

    public static void sleep(int seconds) {
        try {
            TimeUnit.SECONDS.sleep(seconds);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}
```
